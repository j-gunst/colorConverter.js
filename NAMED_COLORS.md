```ts
import colors from '@nawael/color-converter.js/dist/esm/namedColors.json'

type Color = {
  name: string
  hex: string
  rgb: { red: number; green: number; blue: number }
  hsl: { hue: number; saturation: number; lightness: number }
}

const styles = `
    <style>
        .table {width:800px;border-collapse:collapse;margin:0 auto;}
        .table thead {color:gainsboro;}
        .table thead th {padding:10px;text-align:left;width:25%;}
        .table tbody tr {border-bottom:3px solid #383737;}
        .table tbody tr:hover {box-shadow:0 0 8px 0 #fff;}
        .table tbody td {padding:10px;font-family:system-ui;}
    </style>`

const table = `
    <table class="table">
        <thead>
        <tr>
            <th>Name</th>
            <th>Hex</th>
            <th>RGB</th>
            <th>HSL</th>
        </tr>
        </thead>
        <tbody></tbody>
    </table>`

/**
 * Determines if the text color should be black or white based on the brightness of the color.
 *
 * @param red {number}
 * @param green {number}
 * @param blue {number}
 * @returns {string}
 */
const blackOrWhite = (red: number, green: number, blue: number) =>
  red * 0.299 + green * 0.587 + blue * 0.114 > 178 ? '#000' : '#fff'

/**
 * Displays the color lines.
 *
 * @param item {Color}
 * @returns {string}
 */
const displayColorLines = (item: Color) => {
  const {
    name,
    hex,
    rgb: { red, green, blue },
    hsl: { hue, saturation, lightness }
  } = item

  const rgb = `rgb(${red}, ${green}, ${blue})`
  const hsl = `hsl(${hue}, ${saturation}%, ${lightness}%)`

  return `
    <tr style="color: ${blackOrWhite(red, green, blue)}">
        <td style="background-color:${name}">${name}</td>
        <td style="background-color:${hex}">${hex}</td>
        <td style="background-color:${rgb}">${rgb}</td>
        <td style="background-color:${hsl}">${hsl}</td>
    </tr>`
}

export const displayColorsValues = () => {
  const content = document.querySelector('.content')

  content && (content.innerHTML = `${styles}${table}`)

  const tbody = document.querySelector('tbody')

  Object.values(colors).forEach((item) => {
    tbody?.insertAdjacentHTML('beforeend', displayColorLines(item))
  })
}
```
